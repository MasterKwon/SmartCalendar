📅 스마트 기간선택

빠른기간선택 버튼과 고급 콜백 시스템을 갖춘 향상된 날짜/기간 선택 컴포넌트

🚀 주요 개선사항

  • 함수 시그니처 변경: openDatePicker(start, end, trigger, callbacks)
  • 팝업 내 빠른선택: [오늘/이번주/이번달/4주] 버튼 자동 추가
  • 년/월 직접 이동: 달력 헤더 클릭으로 년도/월 선택기 팝업
  • 4주 모드 감지: 2025.09.01 이후 달력에서 4주 버튼 자동 활성화
  • 고급 콜백: onQuickSelect, onManualSelect, onRangeComplete, onValidationError
  • 데이터 형식: YYYYMMDD (API 연동) + YYYY-MM-DD (화면 표시)
  • 완전 호환: 기존 코드 수정 없이 동작

🗓️ 단일 날짜 선택

선택된 날짜: 날짜를 선택하면 YYYYMMDD, YYYY-MM-DD 형식으로 표시됩니다. 
// 🔵 방법 1: 기존 createSingleDatePicker 방식 (완전 호환)
const singlePicker = createSingleDatePicker({
    onApply: (selectedDate) => {
        console.log('선택된 날짜:', selectedDate);
        // YYYYMMDD 형식으로 변환: formatDate(selectedDate)
    }
});
singlePicker.open(inputElement);

// 🆕 방법 2: 함수 오버로딩 방식 (1개 인자)

// ✅ 두 방식 모두 동일하게 작동하며 기존 코드와 완전 호환됩니다.
// 📅 년/월 직접 이동: 달력 헤더(2025.06) 클릭으로 년도/월 선택기 팝업

📊 스마트 기간선택 (주요 기능)

선택된 기간: 기간을 선택하면 자동 계산되며, API 전송용과 화면 표시용 형식으로 표시됩니다. 
// 🆕 함수 시그니처 변경: openDatePicker(start, end, trigger, callbacks)
openDatePicker('startDate', 'endDate', 'start', {
    // 🎯 팝업 내 빠른선택 버튼 클릭 시 (자동 추가됨)
    onQuickSelect: (startDate, endDate, rangeType, details) => {
        console.log('빠른선택:', rangeType, details.formattedStart, '~', details.formattedEnd);
        // rangeType: 'today', 'thisWeek', 'thisMonth', 'fourWeeks'
        // 📊 데이터는 YYYYMMDD 형식으로 처리됨
        if (rangeType === 'fourWeeks') {
            // 4주 버튼: 4주 모드 달력일 때만 활성화
            console.log('4주 기간:', details.formattedStart, '~', details.formattedEnd);
        }
    },
    
    // 👆 개별 날짜 선택 시  
    onManualSelect: (selectedDate, inputType, details) => {
        console.log('개별선택:', inputType, details.formattedDate);
        // inputType: 'start' 또는 'end'
        // 미래/과거 날짜 체크, 업무일 검증 등
    },
    
    // ✅ 기간 선택 완료 시 (자동 트리거)
    onRangeComplete: (startDate, endDate, details) => {
        console.log('기간완성:', details.durationDays + '일간');
        // API 호출, 데이터 조회, 장기간 경고 등
        const apiData = {
            start: details.formattedStart.replace(/-/g, ''), // YYYYMMDD
            end: details.formattedEnd.replace(/-/g, '')     // YYYYMMDD
        };
    },
    
    // ❌ 유효성 검사 실패 시
    onValidationError: (errorType, message, details) => {
        console.error('검증실패:', errorType, message);
        // 시작일 > 종료일, 날짜 형식 오류 등
    }
});

// 📝 함수 오버로딩: 기존 방식도 호환 (1개 인자)
openDatePicker('singleDate'); // 단일 날짜 선택
